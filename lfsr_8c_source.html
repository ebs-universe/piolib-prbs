91 int lfsr16_bGetNextBit(lfsr16_t * lfsrp){
92  int lsb = *lfsrp & 1;
93  *lfsrp >>= 1; /* Shift register */
94  if (lsb == 1){ /* Only apply toggle mask if output bit is 1*/
95  *lfsrp ^= LFSR16_TAPS; /* Apply toggle mask, value has 1 at bits corresponding to taps, 0 elsewhere. */
96  }
97  return lsb;
98 }
